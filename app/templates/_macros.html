{# Responsive : WebP srcset (+ JPEG fallback) with intrinsic width/height from the manifest so images never shift layout. Import with context so IMG / IMG_WIDTHS globals are visible. #} {% macro resp_img(key, alt, sizes, klass='', priority=false, position='') %} {%- set dims = IMG.get(key, {'w': 1600, 'h': 1067}) -%} {%- set widths = IMG_WIDTHS | select('le', dims.w) | list -%} {%- if not widths -%}{%- set widths = [dims.w] -%}{%- endif -%} {% endmacro %}
